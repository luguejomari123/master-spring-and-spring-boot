mysql:8-oracle      = image
--env MYSQL_ROOT_PASSWORD=dummypassword     = root password
--env MYSQL_USER=todos-user         = user id
--env MYSQL_PASSWORD=dummytodos    = user password
--env MYSQL_DATABASE=todos    = database name
--name mysql    = container name
--publish 3306:3306  = port to publish
